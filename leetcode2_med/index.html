2. Add Two Numbers

You are given two linked lists representing two non-negative numbers. The digits are stored in reverse order and each of their nodes contain a single digit. Add the two numbers and return it as a linked list.

Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)
Output: 7 -> 0 -> 8

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

/**
 * Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)
 * Output: 7 -> 0 -> 8
 * Explanation: 342 + 465 = 807.
 */

public class leetcode2_addTwoNumbers {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode dummy = new ListNode(0);
        ListNode p1 = l1, p2 = l2;
        ListNode cur = dummy;
        int res = 0;
        //int temp = 0;

        while ((p1 != null) || (p2 != null)){
            if (p1 != null) {
                res = res + p1.val;
                p1 = p1.next;
            }
            if (p2 != null) {
                res = res + p2.val;
                p2 = p2.next;
            }

            cur.next = new ListNode(res % 10);
            //cur.val = res % 10;
            res = res / 10;
            cur = cur.next;
        }

        if(res != 0) {
            cur.next = new ListNode(res);
        }
        return dummy.next;
    }

    public static void main(String[] args ){
        leetcode2_addTwoNumbers l = new leetcode2_addTwoNumbers();
        ListNode input1 = new ListNode(2);
        input1.next = new ListNode(4);
        input1.next.next = new ListNode(3);

        ListNode input2 = new ListNode(5);
        input2.next = new ListNode(6);
        input2.next.next = new ListNode(4);

        ListNode res = new ListNode(0);
        res = l.addTwoNumbers(input1, input2);

        while (res != null) {
            System.out.print(res.val + "\n");
            res = res.next;
        }

    }
}
