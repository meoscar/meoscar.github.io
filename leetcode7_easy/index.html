7. Reverse Integer

Given a 32-bit signed integer, reverse digits of an integer.

Example 1:

Input: 123
Output: 321

Example 2:

Input: -123
Output: -321

Example 3:

Input: 120
Output: 21

Note:
Assume we are dealing with an environment which could only store integers within the 32-bit signed integer range: [−231,  231 − 1]. For the purpose of this problem, assume that your function returns 0 when the reversed integer overflows.

 
 



public class leetcode7_ReverseInteger {


/*

Given a 32-bit signed integer, reverse digits of an integer.

Example 1:

Input: 123
Output: 321

Example 2:

Input: -123
Output: -321

Example 3:

Input: 120
Output: 21

Note:
Assume we are dealing with an environment which could only store integers within the 32-bit signed integer range: [−2^31,  2^31 − 1]. For the purpose of this problem, assume that your function returns 0 when the reversed integer overflows.

*/


    // Method 1
    public int reverse(String input) {
        int sign = 1;
        long res = 0;

        String newStr = "";
        for (int i = input.length(); i > 0; i--) {
            //System.out.printf(input.length() + "," + i + "\n");
            if (i == 1 && input.charAt(0) == '-')
                newStr = '-' + newStr;
            else
                newStr += input.charAt(i - 1);
        }

        try {
            res = Long.parseLong(newStr);
            if (res > Integer.MAX_VALUE || res < Integer.MIN_VALUE)
                return 0;
        } catch (NumberFormatException e) {
            System.out.printf("error\n");
            return 0;
        }

        return (int) res;
    }

    // Method 2
    public int reverse(int input) {
        int sign = 1;
        long absIn = input;
        long res = 0;

        if (input < 0) {
            sign = -1;
            absIn = -input;
        }

        while (absIn != 0) {
            int temp = (int) absIn % 10;
            res = res * 10 + temp;
            absIn /= 10;
        }

     
        res *= sign;
        if (res > Integer.MAX_VALUE || res < Integer.MIN_VALUE) // MAX_VALUE = 2147483647; MIN_VALUE = -2147483648;
            return 0;


        return (int) res;
    }

    public static void main(String[] args) {
        int out = 0;
        leetcode7_ReverseInteger in = new leetcode7_ReverseInteger();
        // method 1
        int input = 123456789;

        out = in.reverse(input);
        System.out.printf("input: " + input + " -> result: " + out + "\n");
        // method 2
        // out = in.reverse(Integer.parseInt(input));
        // System.out.printf("input: " + input + " -> result: " + out + "\n");
    }

}
